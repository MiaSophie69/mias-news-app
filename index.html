    <script>

        let allNews = [];

        let favoritenSet = new Set(JSON.parse(localStorage.getItem('mias_favoriten') || '[]'));

        let aktuelleKategorie = 'Alle';

        let nurFavoritenModus = false;


        // Filtert unleserliche Zeichen aus den API-Texten heraus

        function bereinigeText(text) {

            if (!text) return "";

            return text

                .replace(/&amp;/g, "&")

                .replace(/&lt;/g, "<")

                .replace(/&gt;/g, ">")

                .replace(/Ã¤/g, "ä")

                .replace(/Ã¶/g, "ö")

                .replace(/Ã¼/g, "ü")

                .replace(/Ã„/g, "Ä")

                .replace(/Ã–/g, "Ö")

                .replace(/Ãœ/g, "Ü")

                .replace(/ÃŸ/g, "ß")

                .replace(/Â/g, "");

        }


        async function ladeEchteNews() {

            const container = document.getElementById('news-container');

            container.innerHTML = "<div class='empty-state'>Lade echte Live-News über Mias Server...</div>";

            

            try {

                // Deine echte Render-Adresse ist jetzt hier fest eingebaut!

                const response = await fetch(`https://mias-news-backend.onrender.com/api/live-news`);

                const data = await response.json();

                

                if (data.articles) {

                    allNews = data.articles.filter(a => a.title && a.description).map((artikel, index) => {

                        let kat = "Weltgeschehen";

                        let t = artikel.title.toLowerCase();

                        if (t.includes("musik") || t.includes("film") || t.includes("kino") || t.includes("star") || t.includes("album") || t.includes("konzert")) kat = "Popkultur";

                        else if (t.includes("ki") || t.includes("apple") || t.includes("google") || t.includes("tech") || t.includes("smartphone") || t.includes("chatgpt")) kat = "Technologie";

                        

                        return {

                            id: 'news_' + index,

                            titel: bereinigeText(artikel.title),

                            beschreibung: bereinigeText(artikel.description),

                            kategorie: kat,

                            quelle: bereinigeText(artikel.source.name),

                            publikationen: Math.floor(Math.random() * 36) + 12

                        };

                    });

                }

                

                if(allNews.length === 0) nutzeFallback();

            } catch (e) {

                console.error("Fehler beim Laden:", e);

                nutzeFallback();

            }

            renderNews();

        }


        function nutzeFallback() {

            allNews = [

                { id: 'f1', titel: "Mias Web-App l&auml;uft: Gro&szlig;es Feature-Update steht kurz bevor!", beschreibung: "Nach intensiven Programmier-Sessions steht das neue Grundger&uuml;st der App als PWA auf dem iPhone Homescreen.", kategorie: "Technologie", quelle: "Mias Tech-Blog", publikationen: 45 },

                { id: 'f2', titel: "Popkultur-Sensation: Rekorde pur im neuen Musik-Jahr.", beschreibung: "Die gr&ouml;&szlig;ten Stars k&uuml;ndigen &uuml;berraschend neue Alben und gigantische Welttouren an.", kategorie: "Popkultur", quelle: "Rolling Stone", publikationen: 32 },

                { id: 'f3', titel: "Gro&szlig;e Entdeckungen im Weltraum fesseln die Forschung.", beschreibung: "Wissenschaftler haben faszinierende neue Daten ausgewertet, die unseren Blick auf das Universum ver&auml;ndern.", kategorie: "Weltgeschehen", quelle: "Science Daily", publikationen: 22 }

            ];

        }


        function renderNews() {

            const container = document.getElementById('news-container');

            const searchWord = document.getElementById('search-input').value.toLowerCase();

            container.innerHTML = '';

            

            let gefiltert = allNews;

            

            if (nurFavoritenModus) {

                gefiltert = gefiltert.filter(n => favoritenSet.has(n.id));

            } else if (aktuelleKategorie !== 'Alle') {

                gefiltert = gefiltert.filter(n => n.kategorie === aktuelleKategorie);

            }

            

            if (searchWord && !nurFavoritenModus) {

                gefiltert = gefiltert.filter(n => n.titel.toLowerCase().includes(searchWord) || n.beschreibung.toLowerCase().includes(searchWord));

            }

            

            if (gefiltert.length === 0) {

                container.innerHTML = nurFavoritenModus ? 

                    `<div class='empty-state'> Keine Favoriten gespeichert.</div>` : 

                    `<div class='empty-state'>Keine Treffer f&uuml;r deine Suche.</div>`;

                return;

            }

            

            gefiltert.forEach(news => {

                const istFav = favoritenSet.has(news.id);

                const classKat = news.kategorie.toLowerCase();

                const card = document.createElement('div');

                card.className = 'card';

                card.innerHTML = `

                    <div class="card-top">

                        <span class="badge ${classKat}">${news.kategorie}</span>

                        <button class="card-heart ${istFav ? 'active' : ''}" onclick="toggleFavorit('${news.id}', event)">${istFav ? '❤️' : '🤍'}</button>

                    </div>

                    <h2 onclick="this.parentElement.classList.toggle('open')">${news.titel}</h2>

                    <p>${news.beschreibung}<br><br><strong>Hauptquelle:</strong> ${news.quelle}</p>

                    <div class="card-footer">

                        <span>📰 ${news.publikationen} Publikationen</span>

                        <span>Mehr lesen &rarr;</span>

                    </div>

                `;

                container.appendChild(card);

            });

        }


        function filterKategorie(kat, btn) {

            aktuelleKategorie = kat;

            document.querySelectorAll('.cat-btn').forEach(b => b.classList.remove('active'));

            btn.classList.add('active');

            renderNews();

        }


        function toggleFavorit(id, event) {

            event.stopPropagation();

            if (favoritenSet.has(id)) favoritenSet.delete(id);

            else favoritenSet.add(id);

            localStorage.setItem('mias_favoriten', JSON.stringify([...favoritenSet]));

            renderNews();

        }


        function toggleFavoritenModus() {

            nurFavoritenModus = !nurFavoritenModus;

            document.getElementById('app-title').innerText = nurFavoritenModus ? "Deine Favoriten" : "Mias News";

            document.getElementById('controls-area').style.display = nurFavoritenModus ? 'none' : 'block';

            document.getElementById('back-btn').style.display = nurFavoritenModus ? 'flex' : 'none';

            document.getElementById('main-heart-btn').style.display = nurFavoritenModus ? 'none' : 'block';

            renderNews();

        }


        ladeEchteNews();

    </script>
